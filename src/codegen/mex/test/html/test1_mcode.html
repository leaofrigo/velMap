1function p = test(xxCentro,NumOfCellsX,cellDepths,cellDepthsCentro,NumOfCellsY,VelVector,...
  2    Sumnum2,xx,Direcao,C,StartEdge,tracknew1,depthnew1,OpenFiles,XXQuiver,YYQuiver,...
  3    Quiver3VectorU,Quiver3VectorV,Quiver3VectorW,Quiver3VectorMag,Quiver3VectorErr,VelRL,Directory,...
  4    quiv,handles) %%codegen
  5
  6xxCentroLin = linspace(min(xxCentro(:)),max(xxCentro(:)),NumOfCellsX);
  7cellDepthsLin = linspace(min(cellDepths),max(cellDepths),NumOfCellsY);
  8[xxCentroGrid,cellDepthsGrid] = meshgrid(xxCentroLin,cellDepthsLin);
  9
 10AAgridVel = griddata(xxCentro,cellDepthsCentro,VelVector,xxCentroGrid,cellDepthsGrid);
 11
 12Yy = cellDepths(Sumnum2(1:length(Sumnum2)-1));
 13PerimSupX = zeros(1,2*(length(xx)-1));
 14PerimSupY = zeros(1,2*(length(xx)-1));
 15PerimSupY(1:2:end-1) = Yy;
 16PerimSupY(2:2:end) = Yy;
 17PerimSupX(1) = xx(1);
 18PerimSupX(end) = xx(end);
 19PerimSupX(2:2:end-2) = xx(2:end-1);
 20PerimSupX(3:2:end-1) = xx(2:end-1);
 21
 22Yybottom = cellDepths(Sumnum2(2:end)-1);
 23PerimBotY = zeros(1,2*(length(xx)-1));
 24PerimBotY(1:2:end-1) = Yybottom;
 25PerimBotY(2:2:end) = Yybottom;
 26PerimBotX = PerimSupX;
 27
 28for ij = 1:length(PerimBotX)-1    
 29    TFTemp = and(xxCentroGrid(1,:)>=PerimBotX(ij),xxCentroGrid(1,:)<=PerimBotX(ij+1));
 30    TFTemp1 = and(cellDepthsGrid(:,TFTemp)>=PerimBotY(ij),cellDepthsGrid(:,TFTemp)<=PerimSupY(ij));
 31    TFTemp2 = zeros(size(cellDepthsGrid));
 32    TFTemp2(:,TFTemp) = ~TFTemp1;
 33    TFTemp2 = logical(TFTemp2);
 34    cellDepthsGrid(TFTemp2) = nan; 
 35    xxCentroGrid(TFTemp2) = nan;
 36    AAgridVel(TFTemp2) = nan;    
 37end
 38clf
 39p=pcolor(xxCentroGrid,cellDepthsGrid,AAgridVel);
 40set(p,'EdgeColor','none')
 41colorbar('south')
 42h_bar = findobj(gcf,'Tag','Colorbar');
 43set(get(h_bar,'xlabel'),'String', 'Velocity(m/s)');
 44initpos = get(h_bar,'Position');
 45initfontsize = get(h_bar,'FontSize');
 46set(h_bar,'Position',[initpos(1)+initpos(3)/2 initpos(2) initpos(3)/2 initpos(4)/2],...
 47    'FontSize',initfontsize*.75)
 48hold on
 49
 50if handles.run3d == 1
 51    Pos = get(handles.Options3D.plotpostion3d,'UserData');
 52else
 53    Pos = get(handles.Options2D.plotpostion,'UserData');
 54end
 55set(gcf, 'Position', Pos);
 56plot(tracknew1, depthnew1)
 57colormap(flipud(C([1:4:16,14:50,50:2:end-125,...
 58    end-125:end-50,end-50:4:end-30],:))) % make the color bar look like the River Servoyor Live
 59xlim([tracknew1(1),tracknew1(end)])
 60ylim([min(depthnew1)+.1*diff([0,min(depthnew1)]),0]);
 61xlabel('Length (m)')
 62ylabel('Depth (m)')
 63if StartEdge == 1
 64    set(gca,'XDir','reverse')
 65end
 66% if n > 1
 67%     OpenFiles = OpenFiles{n};
 68% end
 69if quiv{1} ==1
 70    hold on
 71   switch quiv{2}
 72        case 'North'
 73            Vel1 = Quiver3VectorV;      
 74        case 'East'
 75            Vel1 = Quiver3VectorU;        
 76        case 'Upstream'
 77            Vel1 = Quiver3VectorW;        
 78        case 'Magnitude'
 79            Vel1 = Quiver3VectorMag;
 80        case 'Error'
 81            Vel1 = Quiver3VectorErr; 
 82        case 'Transverse'
 83            Vel1 = VelRL(:,1);
 84        case 'Longitudinal'     
 85            Vel1 = VelRL(:,2);
 86    end
 87    switch quiv{3}
 88        case 'North'
 89            Vel2 = Quiver3VectorV;      
 90        case 'East'
 91            Vel2 = Quiver3VectorU;        
 92        case 'Upstream'
 93            Vel2 = Quiver3VectorW;        
 94        case 'Magnitude'
 95            Vel2 = Quiver3VectorMag;
 96        case 'Error'
 97            Vel2 = Quiver3VectorErr; 
 98        case 'Transverse'
 99            Vel2 = VelRL(:,1);
100        case 'Longitudinal'     
101            Vel2 = VelRL(:,2);
102    end
103    hold on
104    if handles.run3d == 1;
105        quivScale = get(handles.Options3D.quiv3dOption,'UserData');
106    else
107        quivScale = get(handles.Options2D.quiv2dOption,'UserData');
108    end
109    if StartEdge ==1
110        quiver(XXQuiver,YYQuiver,-Vel1,Vel2,quivScale)
111    else
112        quiver(XXQuiver,YYQuiver,Vel1,Vel2,quivScale)
113    end
114end
115    
116switch Direcao
117    case 'North'
118        title('Velocity North');
119        [LowerVN,UpperVN] = limites(VelVector); 
120        caxis([LowerVN UpperVN]);
121        savefig(gcf,[Directory 'Session ' OpenFiles '\Velocity_North_Smooth.fig']);
122    case 'East'
123        title('Velocity East');
124        [LowerVE,UpperVE] = limites(VelVector); 
125        caxis([LowerVE UpperVE]);
126        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_East_Smooth.fig']);
127    case 'Upstream'
128        title('Velocity Upstream');
129        [LowerVU,UpperVU] = limites(VelVector); 
130        caxis([LowerVU UpperVU]);
131        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_Upstream_Smooth.fig']);
132    case 'Error'
133        title('Error Velocity');
134        [LowerVD,UpperVD] = limites(VelVector); 
135        caxis([LowerVD UpperVD]);
136        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_Upstream_Smooth.fig']);
137    case 'Magnitude'
138        title('Velocity magnitude');
139        [~,UpperVMag,StdMag] = limites(VelVector); 
140        UpperVMag1 = ceil(max(max(VelVector))*10)/10;
141        hold on
142        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_Magnitude_Smooth.fig']);
143    case 'Transverse'
144        [LowerVR,UpperVR] = limites(VelVector); 
145        caxis([LowerVR UpperVR]);
146        title('Velocity transverse')
147        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_Transversal_Smooth.fig']);
148    case 'Longitudinal'
149        title('Velocity Longitudinal')
150        [LowerVL,UpperVL] = limites(VelVector); 
151        caxis([LowerVL UpperVL]);
152        savefig(gcf,[Directory 'Session ' OpenFiles '/Velocity_Longitudinal_Smooth.fig']);
153end
